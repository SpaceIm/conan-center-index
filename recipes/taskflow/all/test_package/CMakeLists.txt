cmake_minimum_required(VERSION 3.1)
project(test_package)

include(${CMAKE_BINARY_DIR}/conanbuildinfo.cmake)
conan_basic_setup()

if(TASKFLOW_GE_250)
    find_package(Taskflow REQUIRED)
else()
    find_package(Cpp-Taskflow REQUIRED)
endif()

if(DEFINED Cpp-Taskflow_VERSION AND Cpp-Taskflow_VERSION LESS_EQUAL "2.2.0")
    set(_CXX_STANDARD 17)
else()
    set(_CXX_STANDARD 14)
endif()

add_executable(${PROJECT_NAME} test_package.cpp)
set_target_properties(${PROJECT_NAME} PROPERTIES CXX_STANDARD ${_CXX_STANDARD})
if(TASKFLOW_GE_250)
    target_link_libraries(${PROJECT_NAME} Taskflow::Taskflow)
else()
    target_link_libraries(${PROJECT_NAME} Cpp-Taskflow::Cpp-Taskflow)
endif()
